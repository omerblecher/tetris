    **Create `src/hooks/useLeaderboard.ts`** — Real-time subscription to the top-10 scores.

    Uses `onSnapshot` with `orderBy('score', 'desc').limit(10)` query on the `scores` collection. The `useEffect` MUST return the unsubscribe function — not doing so causes a listener leak and phantom state updates after unmount. This is identified as Pitfall 3 in the research.

    ```typescript
    import { useEffect, useState } from 'react';
    import { collection, query, orderBy, limit, onSnapshot } from 'firebase/firestore';
    import { db } from '../firebase/config';
    import type { ScoreEntry } from '../firebase/leaderboard';

    export function useLeaderboard(): ScoreEntry[] {
      const [entries, setEntries] = useState<ScoreEntry[]>([]);

      useEffect(() => {
        const q = query(
          collection(db, 'scores'),
          orderBy('score', 'desc'),
          limit(10),
        );

        // onSnapshot returns unsubscribe — MUST return it so React cleans up on unmount
        return onSnapshot(q, (snapshot) => {
          setEntries(
            snapshot.docs.map((d) => d.data() as ScoreEntry),
          );
        });
      }, []);

      return entries;
    }
    ```

    Note: If the app is run without a Firestore index, `onSnapshot` will throw `FAILED_PRECONDITION` with a console link to create the index. This is expected on first run — the executor should check the console and click the link if it appears.

    **Create `src/components/Leaderboard.tsx`** — Side panel leaderboard matching the neon/synthwave aesthetic.

    Per user decisions (locked):
    - Always visible during play — not a modal
    - Each row: rank + name + score (no level, no date)
    - Real-time via onSnapshot (handled by useLeaderboard hook)
    - Current authenticated player's row is highlighted with neon accent color/glow when in top 10
    - Highlight styling is at Claude's discretion (use cyan glow: `var(--color-accent)` with `textShadow`)

    Use existing CSS custom properties from index.css: `--color-accent` (cyan #00f0f0), `--color-text`, `--color-dim`, `--color-border`, `--color-bg`.

    Style with inline styles (consistent with existing GameBoard overlay pattern from Phase 2). Panel container uses the `.side-panel` CSS class if available, or equivalent inline styles matching the right panel look.

    ```typescript
    import { useLeaderboard } from '../hooks/useLeaderboard';
    import { useAuth } from '../contexts/AuthContext';
    import type { CSSProperties } from 'react';

    const panelStyle: CSSProperties = {
      display: 'flex',
      flexDirection: 'column',
      gap: '4px',
      width: '100%',
    };

    const labelStyle: CSSProperties = {
      fontFamily: 'Orbitron, monospace',
      fontSize: '10px',
      letterSpacing: '2px',
      color: 'var(--color-dim)',
      marginBottom: '6px',
    };

    function rowStyle(isCurrentUser: boolean): CSSProperties {
      return {
        display: 'grid',
        gridTemplateColumns: '24px 1fr auto',
        gap: '4px',
        alignItems: 'center',
        padding: '4px 2px',
        fontFamily: 'Orbitron, monospace',
        fontSize: '10px',
        letterSpacing: '1px',
        borderRadius: '2px',
        ...(isCurrentUser
          ? {
              color: 'var(--color-accent)',
              textShadow: '0 0 8px var(--color-accent)',
              background: 'rgba(0, 240, 240, 0.06)',
            }
          : {
              color: 'var(--color-text)',
              opacity: 0.85,
            }),
      };
    }

    export function Leaderboard() {
      const entries = useLeaderboard();
      const { user } = useAuth();

      return (
        <div style={panelStyle}>
          <div style={labelStyle}>GLOBAL TOP 10</div>
          {entries.map((entry, i) => {
            const isCurrentUser = !!user && user.uid === entry.uid;
            return (
              <div key={entry.uid} style={rowStyle(isCurrentUser)}>
                <span style={{ color: 'var(--color-dim)', fontSize: '9px' }}>
                  #{i + 1}
                </span>
                <span
                  style={{
                    overflow: 'hidden',
                    textOverflow: 'ellipsis',
                    whiteSpace: 'nowrap',
                    maxWidth: '90px',
                  }}
                >
                  {entry.displayName}
                </span>
                <span style={{ textAlign: 'right' }}>
                  {entry.score.toLocaleString()}
                </span>
              </div>
            );
          })}
          {entries.length === 0 && (
            <div
              style={{
                color: 'var(--color-dim)',
                fontFamily: 'Orbitron, monospace',
                fontSize: '9px',
                letterSpacing: '1px',
                marginTop: '8px',
                textAlign: 'center',
              }}
            >
              No scores yet.
              <br />
              Be first!
            </div>
          )}
        </div>
      );
    }
    ```
